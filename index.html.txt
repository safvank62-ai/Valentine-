   <script>
       document.addEventListener('DOMContentLoaded', () => {
           const btnYes = document.getElementById('btn-yes');
           const btnNo = document.getElementById('btn-no');
           const errorArea = document.getElementById('error-area');
           const errorText = document.getElementById('error-text');
           const subText = document.getElementById('sub-text');
           const mainContainer = document.getElementById('main-container');
           const successScreen = document.getElementById('success-screen');
           const bgHearts = document.getElementById('bg-hearts');

           let yesScale = 1;
           let noScale = 1;
           let noClicks = 0;

           // 1. Generate Background Hearts
           function createBackgroundHearts() {
               const heartCount = 15;
               for (let i = 0; i < heartCount; i++) {
                   const heart = document.createElement('div');
                   heart.classList.add('heart');
                   heart.style.left = Math.random() * 100 + 'vw';
                   heart.style.animationDuration = (Math.random() * 5 + 5) + 's';
                   heart.style.animationDelay = (Math.random() * 5) + 's';
                   heart.style.opacity = Math.random() * 0.3;
                   bgHearts.appendChild(heart);
               }
           }
           createBackgroundHearts();

           // 2. Handle "No" Click
           btnNo.addEventListener('click', () => {
               noClicks++;

               // Show Error Messages
               errorArea.classList.remove('opacity-0');
               errorText.innerText = "Invalid response";
               subText.innerText = "Please click again ü•∫";
               
               // Trigger Shake Animation (reset to re-trigger)
               errorText.classList.remove('shake');
               void errorText.offsetWidth; // trigger reflow
               errorText.classList.add('shake');

               // Button Transformations
               // Yes grows
               yesScale += 0.25;
               btnYes.style.transform = `scale(${yesScale})`;
               
               // No shrinks
               noScale = Math.max(0.3, noScale - 0.15); // Don't disappear completely
               btnNo.style.transform = `scale(${noScale})`;
               
               // Optional: Make No button slightly transparent as it shrinks
               btnNo.style.opacity = Math.max(0.4, noScale);
           });

           // 3. Handle "Yes" Click
           btnYes.addEventListener('click', () => {
               // Fade out main container
               mainContainer.classList.add('fade-out');
               
               // Wait for fade out, then show success
               setTimeout(() => {
                   mainContainer.style.display = 'none';
                   successScreen.classList.remove('hidden');
                   startSuccessAnimation();
               }, 800);
           });

           // 4. Success Animation Logic
           function startSuccessAnimation() {
               const container = document.getElementById('success-hearts-container');
               
               // Create floating hearts periodically
               setInterval(() => {
                   const heart = document.createElement('div');
                   heart.innerHTML = '‚ù§Ô∏è';
                   heart.classList.add('success-heart');
                   heart.style.left = Math.random() * 100 + '%';
                   heart.style.top = '100%';
                   heart.style.fontSize = (Math.random() * 20 + 20) + 'px';
                   
                   // Randomize animation duration slightly
                   heart.style.animationDuration = (Math.random() * 2 + 3) + 's';
                   
                   container.appendChild(heart);

                   // Cleanup
                   setTimeout(() => {
                       heart.remove();
                   }, 4000);
               }, 300);
           }
       });
